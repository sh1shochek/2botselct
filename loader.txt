-- [[ UTILS ]]
local function urlEncode(s)
	return tostring(s):gsub("[^%w%-_%.~]", function(c)
		return string.format("%%%02X", string.byte(c))
	end)
end

local function trim(s)
	return tostring(s or ""):gsub("^%s+", ""):gsub("%s+$", "")
end

local function getInjectedKey()
	local key = rawget(env, "script_key") or rawget(env, "FROST_KEY")
	key = trim(key)
	if #key > 5 then return key end
	return nil
end

local function getHWID()
	local hwid = ""
	local device = ""
	pcall(function() hwid = gethwid() end)
	pcall(function() device = game:GetService("RbxAnalyticsService"):GetClientId() end)
	if hwid and #tostring(hwid) > 0 then return tostring(hwid) end
	if device and #tostring(device) > 0 then return tostring(device) end
	return tostring(lp.UserId) .. "_" .. tostring(game.PlaceId)
end

local SAVE_FILE = "frost_key.txt"

local function loadSavedKey()
	local ok, data = pcall(readfile, SAVE_FILE)
	if ok and data and #data > 5 then
		return trim(data)
	end
	return nil
end

local function saveKey(key)
	if key and #key > 0 then
		pcall(writefile, SAVE_FILE, key)
	end
end

local function clearKey()
	pcall(delfile, SAVE_FILE)
end

-- [[ SERVER API ]]
local _session = nil
local _licenseUid = nil

local function applyAuthData(key, data)
	if type(data) ~= "table" then return end
	if data.session then _session = data.session end
	_licenseUid = tonumber(data.license_uid or data.licenseUid or _licenseUid)
	env.script_key = key
	env.FROST_KEY = key
	env.FROST_SESSION = _session
	env.FROST_LICENSE_UID = _licenseUid
	env.SWG_LicenseUID = _licenseUid
	env.FROST_AUTH = {
		session = _session,
		key = key,
		licenseUid = _licenseUid,
	}
end

local function getExecutorName()
	local executor = "Unknown"
	pcall(function()
		if identifyexecutor then
			executor = identifyexecutor()
		elseif getexecutorname then
			executor = getexecutorname()
		elseif syn then
			executor = "Synapse X"
		elseif KRNL_LOADED then
			executor = "Krnl"
		elseif is_sirhurt_closure then
			executor = "SirHurt"
		elseif pebc_execute then
			executor = "ProtoSmasher"
		elseif getgenv and getgenv().WRD_LOADED then
			executor = "WeAreDevs"
		elseif getgenv and getgenv().IS_VFLUX then
			executor = "VFlux"
		elseif getgenv and getgenv().IS_SOLARA then
			executor = "Solara"
		elseif getgenv and getgenv().IS_WAVE then
			executor = "Wave"
		end
	end)
	return executor
end

local function getRobloxUserId()
	local uid = ""
	pcall(function()
		uid = tostring(lp.UserId)
	end)
	return uid
end

local function doCheck(key)
	local hwid = getHWID()
	local timestamp = tostring(os.time())
	local placeId = tostring(game.PlaceId)
	local jobId = tostring(game.JobId or "")
	local executor = getExecutorName()
	local robloxUser = getRobloxUserId()

	local discordId = ""
	pcall(function()
		if isfile and isfile("frost_discord.txt") then
			local ok, data = pcall(readfile, "frost_discord.txt")
			if ok and data then discordId = trim(data) end
		end
	end)

	local url = SERVER_URL
		.. "/check"
		.. "?key=" .. urlEncode(key)
		.. "&hwid=" .. urlEncode(hwid)
		.. "&place_id=" .. urlEncode(placeId)
		.. "&job_id=" .. urlEncode(jobId)
		.. "&timestamp=" .. urlEncode(timestamp)
		.. "&executor=" .. urlEncode(executor)
		.. "&roblox_user=" .. urlEncode(robloxUser)
		.. "&guild=" .. urlEncode(GUILD_ID)
		.. (discordId ~= "" and "&discord_id=" .. urlEncode(discordId) or "")

	local ok, response = pcall(function()
		return game:HttpGet(url, true)
	end)
	if not ok then return false, "Connection failed" end

	local ok2, data = pcall(function()
		return HttpService:JSONDecode(response)
	end)
	if not ok2 or type(data) ~= "table" then return false, "Invalid server response" end

	applyAuthData(key, data)
	return data.status == "ok" or data.status == "activated", data.message or data.status, data
end

local function fetchPayload()
	if not _session then return nil end
	local hwid = getHWID()
	local url = SERVER_URL
		.. "/payload"
		.. "?session=" .. urlEncode(_session)
		.. "&hwid=" .. urlEncode(hwid)
		.. "&guild=" .. urlEncode(GUILD_ID)
	local ok, response = pcall(function()
		return game:HttpGet(url, true)
	end)
	if not ok then return nil end
	return response
end

local function downloadFile(url, path)
	if isfile(path) then return true end
	local ok, data = pcall(game.HttpGet, game, url)
	if ok and data and #data > 100 then
		writefile(path, data)
		return true
	end

local saved = getInjectedKey() or loadSavedKey()

local function runAuthAndLoad(key)
	animIn()
	local stages = {
		{0.10, "Initializing",        0.20},
		{0.25, "Connecting to server", 0.30},
		{0.50, "Authenticating",       0.40},
		{0.75, "Verifying HWID",       0.25},
		{0.90, "Loading script",       0.30},
		{1.00, "Complete!",            0.20},
	}
	task.spawn(function()
		if not readyOk then
			setError(readyMsg)
			task.wait(3.5)
			fadeOutAndDestroy()
			return
		end
		for _, s in ipairs(stages) do
			setProgress(s[1], s[2])
			task.wait(s[3])
		end
		local ok, msg, data = doCheck(key)
		if not ok then
			setError(msg or "Auth failed")
			clearKey()
			task.wait(3.5)
			fadeOutAndDestroy(function()
				showKeyInput(runAuthAndLoad)
			end)
			return
		end
		applyAuthData(key, data)
		setSuccess("loaded in " .. string.format("%.2f", tick() - startTime) .. "s")
		task.wait(0.4)
		fadeOutAndDestroy(function()
			local payload = fetchPayload()
			if payload and #payload > 10 and not payload:find("Error:") then
				local fn, err = loadstring(payload)
				if fn then
					task.spawn(fn)
				else
					warn("[frost.vip] Failed to load payload:", err)
				end
			else
				warn("[frost.vip] Payload error or empty")
			end
		end)
	end)
end

if saved and #saved > 5 then
	runAuthAndLoad(saved)
else
	showKeyInput(runAuthAndLoad)
end
